function doGet(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Database"); if (!sheet) return ContentService.createTextOutput(JSON.stringify({error: "Sheet not found"})).setMimeType(ContentService.MimeType.JSON); var data = sheet.getDataRange().getValues(); var result = {}; for (var i = 0; i < data.length; i++) { if(data[i][0]) { result[data[i][0]] = data[i][1]; } } return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); } function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Database"); var params = JSON.parse(e.postData.contents); var key = params.key; var value = params.value; var data = sheet.getDataRange().getValues(); var found = false; // ค้นหาว่ามี Key นี้อยู่แล้วหรือไม่ ถ้ามีให้อัปเดต for (var i = 0; i < data.length; i++) { if (data[i][0] === key) { sheet.getRange(i + 1, 2).setValue(value); found = true; break; } } // ถ้ายังไม่มี ให้เพิ่มบรรทัดใหม่ if (!found) { sheet.appendRow([key, value]); } return ContentService.createTextOutput(JSON.stringify({status: "success"})).setMimeType(ContentService.MimeType.JSON); }
