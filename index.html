function doGet(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("System_DB"); if (!sheet) return ContentService.createTextOutput(JSON.stringify({error: "Sheet not found"})).setMimeType(ContentService.MimeType.JSON); var data = sheet.getDataRange().getValues(); var result = {}; for (var i = 0; i < data.length; i++) { if(data[i][0]) { var key = data[i][0]; var fullString = ""; for(var j = 1; j < data[i].length; j++) { if(data[i][j]) { fullString += data[i][j]; } } result[key] = fullString; } } return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); } function doPost(e) { var lock = LockService.getScriptLock(); try { lock.waitLock(10000); } catch (err) { return ContentService.createTextOutput(JSON.stringify({status: "error", message: "System Busy"})).setMimeType(ContentService.MimeType.JSON); } try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var dbSheet = ss.getSheetByName("System_DB"); if (!dbSheet) { dbSheet = ss.insertSheet("System_DB"); } var params = JSON.parse(e.postData.contents); var key = params.key; var value = params.value; var chunkSize = 45000; var chunks = [key]; for (var i = 0; i < value.length; i += chunkSize) { chunks.push(value.substring(i, i + chunkSize)); } var maxCols = dbSheet.getMaxColumns(); if (chunks.length > maxCols) { dbSheet.insertColumnsAfter(maxCols, chunks.length - maxCols + 2); } var data = dbSheet.getDataRange().getValues(); var foundIndex = -1; for (var i = 0; i < data.length; i++) { if (data[i][0] === key) { foundIndex = i + 1; break; } } if (foundIndex > -1) { dbSheet.getRange(foundIndex, 1, 1, dbSheet.getMaxColumns()).clearContent(); dbSheet.getRange(foundIndex, 1, 1, chunks.length).setValues([chunks]); } else { dbSheet.appendRow(chunks); } if (key === 'egat_drivers') { updateDriversSheet(ss, JSON.parse(value)); } else if (key === 'egat_ot_history') { updateOTHistorySheet(ss, JSON.parse(value)); } return ContentService.createTextOutput(JSON.stringify({status: "success"})).setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService.createTextOutput(JSON.stringify({status: "error", message: err.toString()})).setMimeType(ContentService.MimeType.JSON); } finally { lock.releaseLock(); } } // ----------------------------------------------------------- function updateDriversSheet(ss, driversArray) { var sheet = ss.getSheetByName("รายชื่อพนักงาน"); if (!sheet) { sheet = ss.insertSheet("รายชื่อพนักงาน"); } sheet.clearContents(); var headers = ["ID", "ชื่อ-นามสกุล", "ชื่อเล่น", "เบอร์โทร", "สังกัด", "แผนก"]; if (!driversArray || driversArray.length === 0) { sheet.getRange(1, 1, 1, headers.length).setValues([headers]); return; } var rows = [headers]; for (var i = 0; i < driversArray.length; i++) { var d = driversArray[i]; rows.push([ d.id || "", d.name || "", d.nickname || "", d.phone || "", d.affiliation || "", d.department || "" ]); } sheet.getRange(1, 1, rows.length, headers.length).setValues(rows); } // ----------------------------------------------------------- function updateOTHistorySheet(ss, otArray) { var sheet = ss.getSheetByName("ประวัติOT"); if (!sheet) { sheet = ss.insertSheet("ประวัติOT"); } sheet.clearContents(); // 🟢 สมศักดิ์เพิ่มคอลัมน์ "ลา (วัน)" ให้ตรงนี้ครับ var headers = ["วันที่", "ชื่อ พขร.", "ประเภทวัน", "เวลาเข้า", "เวลาออก", "OT 1.5 (ชม.)", "OT 1 (ชม.)", "OT 3 (ชม.)", "เวลาทำงานรวม (ชม.)", "ขอเบิก OT (ชม.)", "ลา (วัน)", "สถานะ", "แจ้งขาดเอกสาร", "ID อ้างอิง"]; if (!otArray || otArray.length === 0) { sheet.getRange(1, 1, 1, headers.length).setValues([headers]); return; } var rows = [headers]; for (var i = 0; i < otArray.length; i++) { var ot = otArray[i]; var status = ""; if (ot.isLeave) status = "ลาพัก"; else if (ot.isIncomplete) status = "ขาดเอกสาร"; else if (ot.isPending) status = "ร่าง (รอระบุเวลา)"; else status = "บันทึกเวลาแล้ว"; var totalOTMinutes = (ot.ot15 || 0) + (ot.ot1 || 0) + (ot.ot3 || 0); var ot15_hr = (ot.ot15 || 0) / 60; var ot1_hr = (ot.ot1 || 0) / 60; var ot3_hr = (ot.ot3 || 0) / 60; var total_hr = totalOTMinutes / 60; var request_hr = ot.requestOT !== undefined && ot.requestOT !== "" ? ot.requestOT : "-"; // 🟢 ถ้ามีสถานะ isLeave เป็นจริง ให้แปลงค่าเป็นเลข 1 ถ้าไม่ใช่ให้เป็น 0 var leave_day = ot.isLeave ? 1 : 0; rows.push([ ot.date || "", ot.driver || "", ot.type === 'weekday' ? 'วันธรรมดา' : 'วันหยุด', ot.start || "", ot.end || "", ot15_hr, ot1_hr, ot3_hr, total_hr, request_hr, leave_day, // 🟢 ยิงตัวเลขเข้าคอลัมน์ "ลา (วัน)" status, ot.incompleteReason || "-", ot.id || "" ]); } sheet.getRange(1, 1, rows.length, headers.length).setValues(rows); }
